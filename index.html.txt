    <script>
        let currentImgBase64 = "";

        // Tab 控制
        function showTab(id, btn) {
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            window.scrollTo(0, 0);
        }

        // 記帳計算
        function calculate() {
            const input = document.getElementById('calcInput').value || "0";
            try {
                const total = eval(input.replace(/[^-()\d/*+.]/g, ''));
                const rate = 0.94; // 預設匯率
                const twd = Math.round(total * rate);
                document.getElementById('twdResult').innerText = twd;
                return { thb: total, twd: twd };
            } catch(e) { alert("算式錯誤"); return null; }
        }

        // 高品質圖片處理
        function compressAndPreview(input) {
            const file = input.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(e) {
                const img = new Image();
                img.onload = function() {
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');
                    const maxW = 800; // 寬度提升至 800 確保細節
                    const scale = maxW / img.width;
                    canvas.width = maxW;
                    canvas.height = img.height * scale;
                    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                    // 品質提升至 0.8
                    currentImgBase64 = canvas.toDataURL('image/jpeg', 0.8);
                    const preview = document.getElementById('imgPreview');
                    preview.src = currentImgBase64;
                    preview.classList.remove('hidden');
                }
                img.src = e.target.result;
            }
            reader.readAsDataURL(file);
        }

        function saveRecord() {
            const name = document.getElementById('itemName').value;
            const res = calculate();
            if (!name || !res) return alert("請填寫品項名稱與算式");

            const record = { name, thb: res.thb, twd: res.twd, img: currentImgBase64, id: Date.now() };
            const records = JSON.parse(localStorage.getItem('thai_records') || '[]');
            records.unshift(record);
            localStorage.setItem('thai_records', JSON.stringify(records));
            
            // 重設欄位
            document.getElementById('itemName').value = "";
            document.getElementById('calcInput').value = "";
            document.getElementById('imgPreview').classList.add('hidden');
            currentImgBase64 = "";
            renderLedger();
        }

        function renderLedger() {
            const records = JSON.parse(localStorage.getItem('thai_records') || '[]');
            document.getElementById('ledgerItems').innerHTML = records.map(r => `
                <div class="card p-3 flex justify-between items-center">
                    <div class="flex items-center gap-3">
                        ${r.img ? `<img src="${r.img}" class="w-12 h-12 object-cover rounded" onclick="window.open('${r.img}')">` : '<div class="w-12 h-12 bg-gray-100 rounded"></div>'}
                        <div>
                            <div class="font-bold text-sm">${r.name}</div>
                            <div class="text-[10px] text-gray-400">฿${r.thb} ➔ NT$${r.twd}</div>
                        </div>
                    </div>
                    <button onclick="deleteRecord(${r.id})" class="text-gray-300"><i class="fas fa-times"></i></button>
                </div>
            `).join('');
        }

        function deleteRecord(id) {
            let records = JSON.parse(localStorage.getItem('thai_records') || '[]');
            records = records.filter(r => r.id !== id);
            localStorage.setItem('thai_records', JSON.stringify(records));
            renderLedger();
        }

        // 清單邏輯
        const items = ["護照/美金/泰銖", "上網SIM卡", "泰服預約截圖", "長裙/過膝衣物", "防曬乳/太陽眼鏡", "常備藥品", "泳衣", "行動電源/充電線"];
        function renderChecklist() {
            const checked = JSON.parse(localStorage.getItem('thai_check') || '[]');
            document.getElementById('checklistArea').innerHTML = items.map(item => `
                <label class="flex items-center gap-2 border-b pb-2 border-orange-50">
                    <input type="checkbox" onchange="saveCheck()" value="${item}" ${checked.includes(item) ? 'checked' : ''} class="w-5 h-5 accent-orange-500">
                    <span>${item}</span>
                </label>
            `).join('');
        }
        function saveCheck() {
            const checked = Array.from(document.querySelectorAll('#checklistArea input:checked')).map(i => i.value);
            localStorage.setItem('thai_check', JSON.stringify(checked));
        }

        // 備忘錄與連結自動偵測
        function saveMemo() {
            const text = document.getElementById('memo').value;
            localStorage.setItem('thai_memo', text);
            const urls = text.match(/(https?:\/\/[^\s]+)/g) || [];
            document.getElementById('memoLinkArea').innerHTML = urls.map(url => `
                <a href="${url}" target="_blank" class="bg-orange-500 text-white text-[10px] px-3 py-1 rounded-full">點擊連結 <i class="fas fa-external-link-alt"></i></a>
            `).join('');
        }

        window.onload = () => {
            renderLedger();
            renderChecklist();
            const m = localStorage.getItem('thai_memo') || "";
            document.getElementById('memo').value = m;
            saveMemo();
        };
    </script>
